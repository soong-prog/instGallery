// 纹理生成器类
class TextureGenerator {
  static createWoodTexture() {
    const canvas = document.createElement('canvas')
    canvas.width = 256
    canvas.height = 256
    const ctx = canvas.getContext('2d')
 
    // 调亮基底色
    ctx.fillStyle = '#3C2415'  // 稍微调亮的深棕色
    ctx.fillRect(0, 0, 1024, 1024)
 
    // 基础木纹 - 调亮并增加暖色调
    for (let i = 0; i < 40; i++) {
      const y = Math.random() * 1024
      ctx.beginPath()
      ctx.strokeStyle = `rgba(140, 70, 30, ${Math.random() * 0.3 + 0.2})`  // 更暖的棕色
      ctx.lineWidth = Math.random() * 30 + 15
      
      // 不规则曲线
      ctx.moveTo(0, y)
      const cp1y = y + Math.random() * 100 - 50
      const cp2y = y + Math.random() * 100 - 50
      const endy = y + Math.random() * 60 - 30
      ctx.bezierCurveTo(
        340, cp1y,
        680, cp2y,
        1024, endy
      )
      ctx.stroke()
    }
 
    // 年轮纹理 - 调亮颜色
    for (let i = 0; i < 12; i++) {
      const x = Math.random() * 1024
      const y = Math.random() * 1024
      const radius = Math.random() * 100 + 50
      
      // 多层次的年轮
      for (let j = 0; j < 8; j++) {
        const r = radius * (1 - j * 0.1)
        const gradient = ctx.createRadialGradient(x, y, 0, x, y, r)
        gradient.addColorStop(0, `rgba(120, 60, 30, ${0.1 + j * 0.02})`)  // 调亮内圈
        gradient.addColorStop(0.7, `rgba(100, 50, 25, ${0.05 + j * 0.01})`)  // 调亮过渡区
        gradient.addColorStop(1, 'rgba(60, 36, 21, 0)')  // 调亮外圈
        ctx.fillStyle = gradient
        ctx.beginPath()
        ctx.arc(x, y, r, 0, Math.PI * 2)
        ctx.fill()
      }
    }
 
    // 添加暖色调层
    ctx.fillStyle = 'rgba(255, 200, 150, 0.1)'  // 温暖的半透明叠加层
    ctx.fillRect(0, 0, 1024, 1024)
 
    // 表面划痕 - 增加可见度
    for (let i = 0; i < 200; i++) {
      const x = Math.random() * 1024
      const y = Math.random() * 1024
      const length = Math.random() * 100 + 50
      const angle = Math.random() * Math.PI
      
      ctx.save()
      ctx.translate(x, y)
      ctx.rotate(angle)
      ctx.beginPath()
      ctx.strokeStyle = `rgba(255, 255, 255, ${Math.random() * 0.04 + 0.02})`  // 增加亮度
      ctx.lineWidth = Math.random() * 1 + 0.5
      ctx.moveTo(-length/2, 0)
      ctx.lineTo(length/2, 0)
      ctx.stroke()
      ctx.restore()
    }
 
    // 高光效果 - 增强光泽
    for (let i = 0; i < 2000; i++) {
      const x = Math.random() * 1024
      const y = Math.random() * 1024
      const size = Math.random() * 4 + 2
      const gradient = ctx.createRadialGradient(x, y, 0, x, y, size)
      gradient.addColorStop(0, `rgba(255, 245, 235, ${Math.random() * 0.06 + 0.03})`)  // 增加高光强度
      gradient.addColorStop(1, 'rgba(255, 245, 235, 0)')
      ctx.fillStyle = gradient
      ctx.fillRect(x - size, y - size, size * 2, size * 2)
    }
 
    const texture = new THREE.CanvasTexture(canvas)
    texture.wrapS = texture.wrapT = THREE.RepeatWrapping
    texture.repeat.set(3, 3)
    return texture
  }
 
  static createWallTexture(isUpper) {
    const canvas = document.createElement('canvas')
    canvas.width = 512
    canvas.height = 512
    const ctx = canvas.getContext('2d')
 
    this.drawWallBaseColor(ctx, isUpper)
    this.drawWallLargeNoise(ctx, isUpper)
    this.drawWallFineNoise(ctx, isUpper)
    this.drawWallLines(ctx, isUpper)
 
    const texture = new THREE.CanvasTexture(canvas)
    texture.wrapS = texture.wrapT = THREE.RepeatWrapping
    texture.repeat.set(2, 2)
    return texture
  }
 
  // 木地板纹理绘制方法
  static drawWoodBaseColor(ctx) {
    ctx.fillStyle = '#C27C4A'
    ctx.fillRect(0, 0, 256, 256)
  }
 
  static drawWoodPrimaryGrain(ctx) {
    for (let i = 0; i < 30; i++) {
      const y = i * (256/30)
      ctx.beginPath()
      ctx.strokeStyle = `rgba(140, 60, 30, ${Math.random() * 0.3 + 0.4})`
      ctx.lineWidth = Math.random() * 15 + 10
      ctx.moveTo(0, y)
      ctx.bezierCurveTo(
        85, y + Math.random() * 8 - 4,
        170, y + Math.random() * 8 - 4,
        256, y + Math.random() * 8 - 4
      )
      ctx.stroke()
    }
  }
 
  static drawWoodSecondaryGrain(ctx) {
    for (let i = 0; i < 60; i++) {
      const y = i * (256/60)
      ctx.beginPath()
      ctx.strokeStyle = `rgba(180, 100, 60, ${Math.random() * 0.2 + 0.2})`
      ctx.lineWidth = Math.random() * 3 + 2
      ctx.moveTo(0, y)
      ctx.bezierCurveTo(
        85, y + Math.random() * 4 - 2,
        170, y + Math.random() * 4 - 2,
        256, y + Math.random() * 4 - 2
      )
      ctx.stroke()
    }
  }
 
  static drawWoodGlossLines(ctx) {
    for (let i = 0; i < 120; i++) {
      const y = i * (256/120)
      ctx.beginPath()
      ctx.strokeStyle = `rgba(255, 200, 160, ${Math.random() * 0.1 + 0.05})`
      ctx.lineWidth = 1
      ctx.moveTo(0, y)
      ctx.lineTo(256, y + Math.random() * 2 - 1)
      ctx.stroke()
    }
  }
 
  static drawWoodDarkNoise(ctx) {
    for (let i = 0; i < 8000; i++) {
      const x = Math.random() * 256
      const y = Math.random() * 256
      ctx.fillStyle = `rgba(120, 50, 20, ${Math.random() * 0.15})`
      ctx.fillRect(x, y, 2, 1)
    }
  }
 
  static drawWoodLightNoise(ctx) {
    for (let i = 0; i < 5000; i++) {
      const x = Math.random() * 256
      const y = Math.random() * 256
      ctx.fillStyle = `rgba(255, 220, 180, ${Math.random() * 0.05})`
      ctx.fillRect(x, y, 1, 1)
    }
  }
 
  // 墙面纹理绘制方法
  static drawWallBaseColor(ctx, isUpper) {
    ctx.fillStyle = isUpper ? '#F5F5F0' : '#98AB9B'
    ctx.fillRect(0, 0, 512, 512)
  }
 
  static drawWallLargeNoise(ctx, isUpper) {
    const count = isUpper ? 2000 : 1500
    for (let i = 0; i < count; i++) {
      const x = Math.random() * 512
      const y = Math.random() * 512
      const alpha = Math.random() * (isUpper ? 0.06 : 0.05)
      ctx.fillStyle = isUpper 
        ? `rgba(225, 225, 220, ${alpha})`
        : `rgba(152, 171, 155, ${alpha})`
      ctx.fillRect(x, y, 2, 2)
    }
  }
 
  static drawWallFineNoise(ctx, isUpper) {
    const count = isUpper ? 40000 : 30000
    for (let i = 0; i < count; i++) {
      const x = Math.random() * 512
      const y = Math.random() * 512
      const alpha = Math.random() * (isUpper ? 0.03 : 0.02)
      ctx.fillStyle = isUpper 
        ? `rgba(230, 230, 225, ${alpha})`
        : `rgba(160, 170, 160, ${alpha})`
      ctx.fillRect(x, y, 1, 1)
    }
  }
 
  static drawWallLines(ctx, isUpper) {
    const count = isUpper ? 80 : 60
    for (let i = 0; i < count; i++) {
      const y = Math.random() * 512
      ctx.beginPath()
      ctx.strokeStyle = isUpper
        ? `rgba(230, 230, 225, ${Math.random() * 0.08})`
        : `rgba(140, 150, 140, ${Math.random() * 0.07})`
      ctx.lineWidth = isUpper ? 1.5 : 1
      ctx.moveTo(0, y)
      ctx.lineTo(512, y + Math.random() * 2 - 1)
      ctx.stroke()
    }
  }
}
 
// 组件
function Floor() {
  const materialRef = useRef()
  // 创建木质纹理
  const floorTextures = useMemo(() => {
    const baseTexture = TextureGenerator.createWoodTexture()
    const normalMap = TextureGenerator.createWoodTexture()
    
    // 设置纹理重复和旋转
    baseTexture.wrapS = baseTexture.wrapT = THREE.RepeatWrapping
    baseTexture.repeat.set(4, 4)
    baseTexture.rotation = Math.PI / 2  // 旋转90度
    
    normalMap.wrapS = normalMap.wrapT = THREE.RepeatWrapping
    normalMap.repeat.set(4, 4)
    normalMap.rotation = Math.PI / 2  // 法线贴图也需要旋转
    
    return { baseTexture, normalMap }
  }, [])
 
  useFrame((state) => {
    if (materialRef.current) {
      const pointLight = state.scene.children.find(
        child => child.type === 'PointLight'
      )
 
      if (pointLight) {
        const lightPos = pointLight.position
        const heightFactor = lightPos.y / 20
        
        materialRef.current.clearcoat = 0.3 + (heightFactor * 0.7)
        materialRef.current.clearcoatRoughness = 0.6 - (heightFactor * 0.55)
        materialRef.current.envMapIntensity = 1.2 + (heightFactor * 1.0)
      }
    }
  })
 
  return (
    <Box 
      position={[0, -0.5, 0]} 
      args={[20, 0.1, 20]}
      receiveShadow
    >
      <meshPhysicalMaterial 
        ref={materialRef}
        {...MATERIALS.floor}
        map={floorTextures.baseTexture}
        normalMap={floorTextures.normalMap}
        emissiveMap={floorTextures.baseTexture}
      />
    </Box>
  )
}
 
function Wall({ position, rotation }) {
  const { width, height, thickness } = LAYOUT.wall.dimensions
  const upperHeight = height * LAYOUT.wall.ratio.upper
  const lowerHeight = height * LAYOUT.wall.ratio.lower
 
  const upperTexture = useMemo(() => TextureGenerator.createWallTexture(true), [])
  const lowerTexture = useMemo(() => TextureGenerator.createWallTexture(false), [])
 
  return (
    <>
      <Box 
        position={[position[0], 4.5 + (lowerHeight/2), position[2]]} 
        rotation={rotation}
        args={[width, upperHeight, thickness]}
        receiveShadow
      >
        <meshPhysicalMaterial 
          {...MATERIALS.wall.upper}
          map={upperTexture}
        />
      </Box>
      <Box 
        position={[position[0], 4.5 - (upperHeight/2), position[2]]} 
        rotation={rotation}
        args={[width, lowerHeight, thickness]}
        receiveShadow
      >
        <meshPhysicalMaterial 
          {...MATERIALS.wall.lower}
          map={lowerTexture}
        />
      </Box>
    </>
  )
}
 
// 定义一个Ceiling组件
function Ceiling() {
  // 使用useMemo钩子创建一个纹理，只在组件首次渲染时创建
  const texture = useMemo(() => TextureGenerator.createWallTexture(true), []);
 
  // 返回一个Box组件，设置位置、尺寸和接收阴影
  return (
    <Box 
      position={LAYOUT.ceiling.position}
      args={[
        LAYOUT.ceiling.dimensions.width,
        LAYOUT.ceiling.dimensions.thickness,
        LAYOUT.ceiling.dimensions.depth
      ]}
      receiveShadow
    >
      // 使用meshPhysicalMaterial材质，设置纹理和双面可见
      <meshPhysicalMaterial 
        {...MATERIALS.wall.upper}
        map={texture}
        side={THREE.DoubleSide}  // 仅保留这个参数确保双面可见
      />
    </Box>
  )
}

export function Gallery({ setArtworkRefs, zoomedArtworkIndex, onArtworkZoom, anyArtworkZoomed }) {
  const cameraControls = useGalleryCamera()
  const artworkRefs = useRef([])
  
  useFrame((state, delta) => {
    if (typeof cameraControls === 'function') {
      cameraControls(delta)
    } else if (cameraControls && typeof cameraControls.update === 'function') {
      cameraControls.update(delta)
    }
  })
 
  // 将引用传递给父组件
  useEffect(() => {
    if (setArtworkRefs) {
      setArtworkRefs(artworkRefs.current);
    }
  }, [setArtworkRefs]);
 
  return (
    <>
      <BackgroundMusic 
        url="/music/ambient.mp3"
        volume={0.3}
      />
 
      <Ceiling />
      <Floor />
      <Wall position={LAYOUT.wall.positions.back} rotation={LAYOUT.wall.rotations.back} />
      <Wall position={LAYOUT.wall.positions.left} rotation={LAYOUT.wall.rotations.left} />
      <Wall position={LAYOUT.wall.positions.right} rotation={LAYOUT.wall.rotations.right} />
      
      <Sculpture 
        position={[0, 0, -5]}
        modelUrl="/models/sculpture.obj"
      />
      
      <Artwork 
        ref={el => artworkRefs.current[0] = el}
        {...LAYOUT.artworks.central} 
        isCurrentlyZoomed={zoomedArtworkIndex === 0}
        onZoomChange={() => onArtworkZoom(0)}
        anyArtworkZoomed={anyArtworkZoomed}
      />
 
      {LAYOUT.artworks.left.map((artwork, index) => {
        const artworkIndex = index + 1;
        return (
          <Artwork 
            key={`left-${index}`}
            ref={el => artworkRefs.current[artworkIndex] = el}
            {...artwork}
            size={[4, 2.7, 0.1]}
            rotation={[0, -Math.PI/2, 0]}
            isCurrentlyZoomed={zoomedArtworkIndex === artworkIndex}
            onZoomChange={() => onArtworkZoom(artworkIndex)}
            anyArtworkZoomed={anyArtworkZoomed}
          />
        );
      })}
 
      {LAYOUT.artworks.right.map((artwork, index) => {
        const artworkIndex = index + 1 + LAYOUT.artworks.left.length;
        return (
          <Artwork 
            key={`right-${index}`}
            ref={el => artworkRefs.current[artworkIndex] = el}
            {...artwork}
            size={[4, 2.7, 0.1]}
            rotation={[0, Math.PI/2, 0]}
            isCurrentlyZoomed={zoomedArtworkIndex === artworkIndex}
            onZoomChange={() => onArtworkZoom(artworkIndex)}
            anyArtworkZoomed={anyArtworkZoomed}
          />
        );
      })}
 
      <Lighting />
    </>
  )
}
